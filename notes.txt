3. setup GNS3 to connect to local ubuntu host

sudo chown root:richard /dev/net/tun
sudo usermod -a -G richard
sudo tunctl -g netdev -t tap0
sudo ifconfig tap0 192.168.x.x netmask 255.255.255.0 up
sudo ip route add 192.168.x.0/128 dev wlp2s0
